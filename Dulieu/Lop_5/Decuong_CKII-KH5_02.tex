	\item Tích cực dọn dẹp vệ sinh trường, lớp, nhà ở,\ldots
	\item Đổ rác đúng nơi quy định, hạn chế sử dụng túi nilon,\ldots
	\item Tiết kiệm điện, nước trong sinh hoạt.
	\item Tích cực trồng cây xanh,\ldots
\end{itemize}
}
